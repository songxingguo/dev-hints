function generator (name, attr) {
  for (let i = 24; i >= 1; i--) {
    console.log(`.col-${name ? `${name}-` : ''}${i} {
        ${attr}: ${parseFloat(((100 / 24) * i).toFixed(4))}%;
    }`)
  }
}
generator("pull", "right")

    margin-left: ($gutter / -2);
    margin-right: ($gutter / -2);
  }
}
// 自定义列之间有间隔
